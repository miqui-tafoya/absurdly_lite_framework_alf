    public function sendData($data) {
        $new = $this->db->crudCall('create', 'tabla_db', $data, '', '');
        return $new;
    }
    public function updateEntryData($data, $id) {
        $mod = $this->db->crudCall('update', 'tabla_db', $data, $id, 'ID');
        return $mod;
    }
    public function deleteEntryData($id) {
        $del = $this->db->crudCall('delete', 'tabla_db', '', ['ID' => $id], '');
        return $del;
    }
